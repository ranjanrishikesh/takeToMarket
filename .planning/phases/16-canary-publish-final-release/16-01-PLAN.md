EXPECTED_VERSION="0.1.0"

smoke_run() {
  local label="$1"
  local cmd="$2"
  local tmphome
  tmphome=$(mktemp -d -t ttm-smoke-${label}-XXXXXX)
  echo "[${label}] HOME=${tmphome}"
  local attempt=0
  local max=3
  while [ $attempt -lt $max ]; do
    attempt=$((attempt+1))
    echo "[${label}] attempt ${attempt}/${max}"
    OUT=$(HOME="$tmphome" timeout 60 bash -c "$cmd" 2>/tmp/ttm-smoke-${label}-stderr.log)
    EXIT=$?
    echo "[${label}] exit=${EXIT}"
    echo "[${label}] stdout (first 2KB):"; echo "$OUT" | head -c 2048; echo ""
    echo "[${label}] stderr (first 1KB):"; head -c 1024 /tmp/ttm-smoke-${label}-stderr.log
    if [ "$EXIT" -eq 0 ] && echo "$OUT" | grep -qF "$BANNER" && echo "$OUT" | grep -qF "$DRYRUN_LINE"; then
      echo "[${label}] PASS"
      if [ -d "$tmphome/.claude/plugins/taketomarket" ] || [ -d "$tmphome/.codex/plugins/taketomarket" ]; then
        echo "[${label}] FAIL: dry-run leaked files into isolated HOME"
        rm -rf "$tmphome"; return 1
      fi
      rm -rf "$tmphome"; return 0
    fi
    if echo "$OUT$(cat /tmp/ttm-smoke-${label}-stderr.log)" | grep -qE "(E404|ETARGET|No matching version|404 Not Found)"; then
      echo "[${label}] registry propagation lag — sleeping 30s before retry"
      sleep 30; continue
    fi
    echo "[${label}] FAIL: unexpected error mode"
    rm -rf "$tmphome"; return 1
  done
  echo "[${label}] FAIL: exhausted ${max} attempts"
  rm -rf "$tmphome"; return 1
}
